/**
 * Google Apps Script - Switch Monitor Data Logger
 * WITH AUTO-ARCHIVE SYSTEM
 *
 * Architecture:
 *   - "Sheet1" (main/active sheet): holds only TODAY's live data from ESP8266
 *   - Archive sheets named "4 April 2026", "5 April 2026", etc.: hold past days' data
 *   - Daily trigger archives yesterday's rows from Sheet1 into its named sheet, then deletes them
 *   - Dashboard fetches only Sheet1 on default load (fast), and also archive sheets when custom dates used
 */

// ─── CONSTANTS ────────────────────────────────────────────────────────────────
var MAIN_SHEET_NAME = 'Sheet1';   // change if your active sheet has a different name
var HEADERS = ['Date', 'Time', 'Prepress 1', 'Prepress 2', 'Prepress 3', 'Prepress 4', 'Prepress 5'];
var TZ = Session.getScriptTimeZone();

// ─── ROUTING ──────────────────────────────────────────────────────────────────

function doGet(e) {
  try {
    var params = e.parameter || {};

    if (params.action === 'getData') {
      return handleGetData(params.fromDate, params.toDate, params.prepress, params.todayOnly);
    }

    // ESP8266 logging request
    return handleLogData(params);

  } catch (error) {
    Logger.log('Error in doGet: ' + error.toString());
    return ContentService.createTextOutput('Error: ' + error.toString());
  }
}

function doPost(e) {
  try {
    var params;
    if (e.postData && e.postData.contents) {
      params = JSON.parse(e.postData.contents);
    } else if (e.parameter) {
      params = e.parameter;
    } else {
      throw new Error('No data received');
    }

    if (params.action === 'getData') {
      return handleGetData(params.fromDate, params.toDate, params.prepress, params.todayOnly);
    }

    throw new Error('Unknown action');

  } catch (error) {
    Logger.log('Error in doPost: ' + error.toString());
    return ContentService
      .createTextOutput(JSON.stringify({ error: error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// ─── ESP8266 DATA LOGGING ─────────────────────────────────────────────────────

function handleLogData(params) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(MAIN_SHEET_NAME) || ss.getActiveSheet();

  // Setup headers if first run
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(HEADERS);
    sheet.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold');
  }

  var now = new Date();
  var date = Utilities.formatDate(now, TZ, 'yyyy-MM-dd');
  var time = Utilities.formatDate(now, TZ, 'HH:mm:ss');

  var statusParam = params.status;
  if (!statusParam) {
    return ContentService.createTextOutput('Error: No status data received');
  }

  var statuses = statusParam.split(',');
  if (statuses.length !== 5) {
    return ContentService.createTextOutput('Error: Expected 5 status values, got ' + statuses.length);
  }

  sheet.appendRow([date, time].concat(statuses));

  return ContentService.createTextOutput('Success: Data logged at ' + date + ' ' + time);
}

// ─── AUTO-ARCHIVE (runs via daily time-driven trigger) ─────────────────────────

/**
 * Archives rows from the main sheet that belong to yesterday (or any past day).
 * Creates a sheet named like "4 April 2026" for each unique past date found.
 * Deletes those rows from the main sheet after archiving.
 *
 * Set this function as a daily time-driven trigger (e.g. runs at 00:05 AM).
 */
function archivePastDays() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var mainSheet = ss.getSheetByName(MAIN_SHEET_NAME) || ss.getActiveSheet();

  var today = Utilities.formatDate(new Date(), TZ, 'yyyy-MM-dd');
  var data = mainSheet.getDataRange().getValues();

  if (data.length <= 1) return; // only header or empty

  // Group rows by date, separate past rows from today's rows
  var archiveMap = {};   // dateStr → array of row arrays
  var keepRows = [data[0]]; // always keep header

  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    var dateVal = row[0];
    var dateStr = '';

    if (dateVal instanceof Date) {
      dateStr = Utilities.formatDate(dateVal, TZ, 'yyyy-MM-dd');
    } else if (typeof dateVal === 'string') {
      dateStr = dateVal;
    } else {
      keepRows.push(row); // unknown format, keep it
      continue;
    }

    if (dateStr >= today) {
      // Today or future: keep in main sheet
      keepRows.push(row);
    } else {
      // Past date: queue for archiving
      if (!archiveMap[dateStr]) archiveMap[dateStr] = [];
      archiveMap[dateStr].push(row);
    }
  }

  // Write each past date's rows to its own archive sheet
  for (var dateKey in archiveMap) {
    var sheetLabel = formatDateLabel(dateKey); // e.g. "4 April 2026"
    var archiveSheet = ss.getSheetByName(sheetLabel);

    if (!archiveSheet) {
      archiveSheet = ss.insertSheet(sheetLabel);
      archiveSheet.appendRow(HEADERS);
      archiveSheet.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold');
      // Move archive sheets to end so Sheet1 stays first
      ss.setActiveSheet(archiveSheet);
      ss.moveActiveSheet(ss.getNumSheets());
    }

    // Append rows to the archive sheet (avoid duplicates by checking existing data)
    var existingData = archiveSheet.getDataRange().getValues();
    var existingSet = {};
    for (var r = 1; r < existingData.length; r++) {
      var key = existingData[r][0] + '|' + existingData[r][1];
      existingSet[key] = true;
    }

    var rowsToAdd = archiveMap[dateKey].filter(function(row) {
      var key = row[0] + '|' + row[1];
      return !existingSet[key];
    });

    if (rowsToAdd.length > 0) {
      archiveSheet.getRange(
        archiveSheet.getLastRow() + 1, 1,
        rowsToAdd.length, HEADERS.length
      ).setValues(rowsToAdd);
    }

    Logger.log('Archived ' + rowsToAdd.length + ' rows to sheet: ' + sheetLabel);
  }

  // Rewrite main sheet with only today's rows
  mainSheet.clearContents();
  if (keepRows.length > 0) {
    mainSheet.getRange(1, 1, keepRows.length, HEADERS.length).setValues(keepRows);
    mainSheet.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold');
  }

  Logger.log('Archive complete. Main sheet now has ' + (keepRows.length - 1) + ' rows.');
}

/**
 * Converts "2026-04-04" → "4 April 2026"
 */
function formatDateLabel(dateStr) {
  var parts = dateStr.split('-');
  var d = new Date(parseInt(parts[0]), parseInt(parts[1]) - 1, parseInt(parts[2]));
  var months = ['January','February','March','April','May','June',
                'July','August','September','October','November','December'];
  return d.getDate() + ' ' + months[d.getMonth()] + ' ' + d.getFullYear();
}

/**
 * Call this once manually to set up the daily archive trigger.
 * After running, you can delete this function or leave it harmlessly.
 */
function createDailyArchiveTrigger() {
  // Remove any existing triggers for archivePastDays to avoid duplicates
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === 'archivePastDays') {
      ScriptApp.deleteTrigger(t);
    }
  });

  // Create a new daily trigger at 12:05 AM
  ScriptApp.newTrigger('archivePastDays')
    .timeBased()
    .everyDays(1)
    .atHour(0)
    .nearMinute(5)
    .create();

  Logger.log('Daily archive trigger created successfully.');
}

// ─── DATA FETCHING ─────────────────────────────────────────────────────────────

/**
 * Main data handler.
 * - todayOnly === 'true': only reads Sheet1 (fast path for dashboard default load)
 * - Otherwise: determines which sheets to read based on date range and queries all of them
 */
function handleGetData(fromDate, toDate, prepressNum, todayOnly) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var allRows = [];

    if (todayOnly === 'true') {
      // Fast path: only read main sheet
      var mainSheet = ss.getSheetByName(MAIN_SHEET_NAME) || ss.getActiveSheet();
      allRows = getRowsFromSheet(mainSheet);
    } else {
      // Custom date range: figure out which sheets to read
      var sheetsToRead = getSheetsForDateRange(ss, fromDate, toDate);
      sheetsToRead.forEach(function(sheet) {
        allRows = allRows.concat(getRowsFromSheet(sheet));
      });
    }

    if (allRows.length === 0) {
      return ContentService
        .createTextOutput(JSON.stringify({ data: [] }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    var result = processSessions(allRows, fromDate, toDate, parseInt(prepressNum));

    Logger.log('Found ' + result.length + ' sessions');

    return ContentService
      .createTextOutput(JSON.stringify({ data: result }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    Logger.log('Error in handleGetData: ' + error.toString());
    return ContentService
      .createTextOutput(JSON.stringify({ error: error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * Returns the list of sheets that may contain data for the given date range.
 * Always includes main sheet (for today/ongoing data) + matching archive sheets.
 */
function getSheetsForDateRange(ss, fromDate, toDate) {
  var sheets = [];
  var today = Utilities.formatDate(new Date(), TZ, 'yyyy-MM-dd');
  var allSheets = ss.getSheets();

  // Always include main sheet if date range overlaps today
  if (toDate >= today) {
    var mainSheet = ss.getSheetByName(MAIN_SHEET_NAME);
    if (mainSheet) sheets.push(mainSheet);
  }

  // Iterate through all archive sheets and include those whose date is in range
  for (var i = 0; i < allSheets.length; i++) {
    var s = allSheets[i];
    var name = s.getName();
    if (name === MAIN_SHEET_NAME) continue;

    var sheetDate = parseDateLabel(name); // "4 April 2026" → "2026-04-04"
    if (sheetDate && sheetDate >= fromDate && sheetDate <= toDate) {
      sheets.push(s);
    }
  }

  return sheets;
}

/**
 * Converts "4 April 2026" → "2026-04-04", returns null if not a valid date label
 */
function parseDateLabel(label) {
  try {
    var months = {
      'January':'01','February':'02','March':'03','April':'04',
      'May':'05','June':'06','July':'07','August':'08',
      'September':'09','October':'10','November':'11','December':'12'
    };
    var parts = label.trim().split(' ');
    if (parts.length !== 3) return null;
    var month = months[parts[1]];
    if (!month) return null;
    var day = parts[0].length === 1 ? '0' + parts[0] : parts[0];
    return parts[2] + '-' + month + '-' + day;
  } catch (e) {
    return null;
  }
}

/**
 * Extracts all data rows (excluding header) from a sheet as normalized objects
 */
function getRowsFromSheet(sheet) {
  var data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];

  var rows = [];
  for (var i = 1; i < data.length; i++) {
    var row = data[i];
    var dateVal = row[0];
    var timeVal = row[1];

    var dateStr = '';
    if (dateVal instanceof Date) {
      dateStr = Utilities.formatDate(dateVal, TZ, 'yyyy-MM-dd');
    } else if (typeof dateVal === 'string') {
      dateStr = dateVal;
    } else {
      continue;
    }

    var timeStr = '';
    if (timeVal instanceof Date) {
      timeStr = Utilities.formatDate(timeVal, TZ, 'HH:mm:ss');
    } else if (typeof timeVal === 'string') {
      timeStr = timeVal;
    } else {
      var totalSeconds = Math.round(timeVal * 24 * 60 * 60);
      var h = Math.floor(totalSeconds / 3600);
      var m = Math.floor((totalSeconds % 3600) / 60);
      var s = totalSeconds % 60;
      timeStr = pad(h) + ':' + pad(m) + ':' + pad(s);
    }

    rows.push({
      dateStr: dateStr,
      timeStr: timeStr,
      datetime: dateStr + ' ' + timeStr,
      cols: row
    });
  }

  // Sort rows by datetime (important when merging multiple sheets)
  rows.sort(function(a, b) {
    return a.datetime.localeCompare(b.datetime);
  });

  return rows;
}

/**
 * Processes normalized rows into sessions with duration and idle time.
 * prepressCol: 2 = Prepress 1, 3 = Prepress 2, ... (matches spreadsheet column index)
 */
function processSessions(rows, fromDate, toDate, prepressCol) {
  var result = [];
  var currentSession = null;

  for (var i = 0; i < rows.length; i++) {
    var row = rows[i];

    if (row.dateStr < fromDate || row.dateStr > toDate) continue;

    var status = row.cols[prepressCol];
    var datetime = row.datetime;

    if (status === 'closed') {
      if (!currentSession) {
        currentSession = { closedFrom: datetime, closedTo: datetime };
      } else {
        currentSession.closedTo = datetime;
      }
    } else if (status === 'open' && currentSession) {
      var durationMs = calculateDuration(currentSession.closedFrom, currentSession.closedTo);
      result.push({
        closedFrom: currentSession.closedFrom,
        closedTo: currentSession.closedTo,
        duration: formatDuration(durationMs),
        durationMs: durationMs
      });
      currentSession = null;
    }
  }

  // Handle ongoing session
  if (currentSession) {
    var durationMs = calculateDuration(currentSession.closedFrom, currentSession.closedTo);
    result.push({
      closedFrom: currentSession.closedFrom,
      closedTo: currentSession.closedTo + ' (ongoing)',
      duration: formatDuration(durationMs),
      durationMs: durationMs
    });
  }

  // Calculate idle times between sessions
  for (var i = 0; i < result.length - 1; i++) {
    var currentEnd = result[i].closedTo.replace(' (ongoing)', '');
    var nextStart = result[i + 1].closedFrom;
    var idleMs = calculateDuration(currentEnd, nextStart);
    result[i].idle = formatDuration(idleMs);
    result[i].idleMs = idleMs;
  }

  if (result.length > 0 && !result[result.length - 1].idle) {
    result[result.length - 1].idle = '-';
    result[result.length - 1].idleMs = 0;
  }

  return result;
}

// ─── HELPERS ──────────────────────────────────────────────────────────────────

function calculateDuration(fromDatetime, toDatetime) {
  try {
    var fromParts = fromDatetime.split(' ');
    var toParts = toDatetime.split(' ');
    var fd = fromParts[0].split('-');
    var ft = fromParts[1].split(':');
    var td = toParts[0].split('-');
    var tt = toParts[1].split(':');

    var from = new Date(+fd[0], +fd[1]-1, +fd[2], +ft[0], +ft[1], +ft[2]);
    var to   = new Date(+td[0], +td[1]-1, +td[2], +tt[0], +tt[1], +tt[2]);

    var ms = to.getTime() - from.getTime();
    return ms < 0 ? 0 : ms;
  } catch (e) {
    return 0;
  }
}

function formatDuration(ms) {
  if (ms < 0 || isNaN(ms)) ms = 0;
  var seconds = Math.floor(ms / 1000);
  var minutes = Math.floor(seconds / 60);
  var hours   = Math.floor(minutes / 60);
  var days    = Math.floor(hours / 24);

  hours   = hours % 24;
  minutes = minutes % 60;
  seconds = seconds % 60;

  var parts = [];
  if (days    > 0) parts.push(days    + 'd');
  if (hours   > 0) parts.push(hours   + 'h');
  if (minutes > 0) parts.push(minutes + 'm');
  if (seconds > 0 || parts.length === 0) parts.push(seconds + 's');

  return parts.join(' ');
}

function pad(num) {
  return (num < 10 ? '0' : '') + num;
}
